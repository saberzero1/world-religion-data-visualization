def customLegend(fig, nameSwap):
    for i, dat in enumerate(fig.data):
        for elem in dat:
            if elem == 'name':
                fig.data[i].name = nameSwap[fig.data[i].name]
    return(fig)
def f(x):
    national_stats_highest = national_stats[["chrstgenpct", "judgenpct", "islmgenpct", "budgenpct", "hindgenpct", "zorogenpct", "sikhgenpct", "shntgenpct", "bahgenpct", "taogenpct", "jaingenpct", "confgenpct", "syncgenpct", "anmgenpct", "nonreligpct", "othrgenpct"]]
    max_value_national_inter = national_stats_highest.idxmax(axis=1, skipna=True)
    max_value_national = pd.concat([national_stats, max_value_national_inter], axis=1, join="inner").rename(columns={0:"Largest religion"}).sort_values("Largest religion").query("year==@x")

                        color="Largest religion",
